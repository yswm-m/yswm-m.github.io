using namespace std;
const int N = 1e6+1;
int q[N],ans,n;
bool st[N];
void fun()
{
    for(int i=2;i<=n;i++)
    {
        if(!st[i])
        {
            q[ans++]=i;//жҠҠзҙ ж•°еӯҳиө·жқҘ
            for(int j=i;j<=n;j+=i) //дёҚз®ЎжҳҜеҗҲж•°иҝҳжҳҜиҙЁж•°пјҢйғҪз”ЁжқҘзӯӣжҺүеҗҺйқўе®ғзҡ„еҖҚж•°
            {

            }
        }
    }
}

void get_primes(){
    //еӨ–еұӮд»Һ2~nиҝӯд»ЈпјҢеӣ дёәиҝҷжҜ•з«ҹз®—зҡ„жҳҜ1~nдёӯиҙЁж•°зҡ„дёӘж•°пјҢиҖҢдёҚжҳҜжҹҗдёӘж•°жҳҜдёҚжҳҜиҙЁж•°зҡ„еҲӨе®ҡ
    for(int i=2;i<=n;i++){
        if(!st[i]) primes[cnt++]=i;
        for(int j=0;primes[j]<=n/i;j++){//primes[j]<=n/i:еҸҳеҪўдёҖдёӢеҫ—еҲ°вҖ”вҖ”primes[j]*i<=n,жҠҠеӨ§дәҺnзҡ„еҗҲж•°йғҪзӯӣдәҶе°ұ
        //жІЎе•Ҙж„Ҹд№үдәҶ
            st[primes[j]*i]=true;//з”ЁжңҖе°ҸиҙЁеӣ еӯҗеҺ»зӯӣеҗҲж•°

            //1)еҪ“i%primes[j]!=0ж—¶,иҜҙжҳҺжӯӨж—¶йҒҚеҺҶеҲ°зҡ„primes[j]дёҚжҳҜiзҡ„иҙЁеӣ еӯҗпјҢйӮЈд№ҲеҸӘеҸҜиғҪжҳҜжӯӨж—¶зҡ„primes[j]<iзҡ„
            //жңҖе°ҸиҙЁеӣ еӯҗ,жүҖд»Ҙprimes[j]*iзҡ„жңҖе°ҸиҙЁеӣ еӯҗе°ұжҳҜprimes[j];
            //2)еҪ“жңүi%primes[j]==0ж—¶,иҜҙжҳҺiзҡ„жңҖе°ҸиҙЁеӣ еӯҗжҳҜprimes[j],еӣ жӯӨprimes[j]*iзҡ„жңҖе°ҸиҙЁеӣ еӯҗд№ҹе°ұеә”иҜҘжҳҜ
            //prime[j]пјҢд№ӢеҗҺжҺҘзқҖз”Ёst[primes[j+1]*i]=trueеҺ»зӯӣеҗҲж•°ж—¶пјҢе°ұдёҚжҳҜз”ЁжңҖе°ҸиҙЁеӣ еӯҗеҺ»жӣҙж–°дәҶ,еӣ дёәiжңүжңҖе°Ҹ
            //иҙЁеӣ еӯҗprimes[j]<primes[j+1],жӯӨж—¶зҡ„primes[j+1]дёҚжҳҜprimes[j+1]*iзҡ„жңҖе°ҸиҙЁеӣ еӯҗпјҢжӯӨж—¶е°ұеә”иҜҘ
            //йҖҖеҮәеҫӘзҺҜпјҢйҒҝе…Қд№ӢеҗҺйҮҚеӨҚиҝӣиЎҢзӯӣйҖүгҖӮ
            if(i%primes[j]==0) break;
        }
    }

}
